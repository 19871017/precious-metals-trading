### 3. 严重程度计算

```typescript
private calculateSeverity(
  value: number,
  threshold: number,
  criticalThreshold: number
): 'LOW' | 'MEDIUM' | 'HIGH' | 'CRITICAL' {
  if (value >= criticalThreshold) {
    return 'CRITICAL';
  } else if (value >= threshold * 1.5) {
    return 'HIGH';
  } else if (value >= threshold) {
    return 'MEDIUM';
  } else {
    return 'LOW';
  }
}
```

### 4. 阻塞事件记录

```typescript
private async logBlockEvent(event: BlockEventData): Promise<void> {
  // 记录到数据库
  await query(`
    INSERT INTO block_event_log (event_type, timestamp, threshold, actual_value, severity, details)
    VALUES ($1, $2, $3, $4, $5, $6)
  `, [
    event.eventType,
    new Date(event.timestamp),
    event.threshold,
    event.actualValue,
    event.severity,
    JSON.stringify(event.details),
  ]);

  // 如果启用了自动上传，上传到 Git
  if (this.config.enableAutoUpload) {
    await this.uploadToGit(event);
  }
}
```

## 使用示例

### 1. 在订单创建时使用 Watchdog

```typescript
// src/routes/api.ts
import { watchdogService } from '../services/WatchdogService';

router.post('/order/create', async (req, res) => {
  const { userId, productCode, ... } = req.body;

  // 开始跟踪订单处理
  const orderId = await createOrderId();
  watchdogService.startTrackingOrder(orderId, userId, productCode);

  try {
    // 风控检查
    watchdogService.updateOrderPhase(orderId, 'VALIDATION');
    await validateOrder(...);

    // 订单撮合
    watchdogService.updateOrderPhase(orderId, 'MATCHING');
    await matchOrder(...);

    // 发送通知
    watchdogService.updateOrderPhase(orderId, 'NOTIFICATION');
    await sendNotification(...);

    res.json({ success: true, orderId });
  } catch (error) {
    res.status(500).json({ error: '订单创建失败' });
  } finally {
    // 停止跟踪
    watchdogService.stopTrackingOrder(orderId);
  }
});
```
